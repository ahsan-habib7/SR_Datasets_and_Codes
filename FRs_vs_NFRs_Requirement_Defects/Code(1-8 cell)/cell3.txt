def clean_text(text):
    text = text.lower()
    text = re.sub(r'[^a-z0-9\s]', ' ', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text


# Ontology rules for auto-detecting defects

    'reliable': 'No measurable reliability metric defined (e.g. 99.9% uptime).',

CONTRADICTION_PATTERNS = [
    (r'simultan\w+', 'Contradiction - simultaneous conflicting actions.'),
    (r'indefinitely.{0,50}(delet|remov|purg)', 'Contradiction - cannot retain and delete simultaneously.'),
    (r'always available.{0,50}(downtime|maintenance)', 'Contradiction - always available conflicts with downtime.'),
    (r'encrypt.{0,50}plain.?text', 'Contradiction - encryption contradicts plain-text storage.'),
    (r'stateless.{0,50}session\s+state', 'Contradiction - stateless cannot maintain session state.'),
    (r'never store.{0,40}re.?auth', 'Contradiction - cannot use passwords that are never stored.'),
    (r'zero.{0,10}latency.{0,30}global', 'Contradiction - zero latency globally is physically impossible.'),
]

SEV_HIGH   = ['security','critical','data loss','crash','authentication',
              'encrypt','compliance','penetration','authorization','payment']
SEV_MEDIUM = ['performance','usability','backup','availability','scalability',
              'response','latency','throughput','recovery','sla','api']


def detect_defect(text):
    tl = text.lower().strip()
    for pattern, note in CONTRADICTION_PATTERNS:
        if re.search(pattern, tl):
            return 'Yes', 'Incorrect Requirement', 'High', note
    for term, explanation in VAGUE_TERMS.items():
        if re.search(r'\b' + re.escape(term) + r'\b', tl):
            sev = 'High' if any(s in tl for s in SEV_HIGH) else \
                  'Medium' if any(s in tl for s in SEV_MEDIUM) else 'Low'
            return 'Yes', 'Ambiguity', sev, "Ambiguity - '" + term + "': " + explanation
    if len(text.split()) < 7:
        return 'Yes', 'Missing Requirement', 'Medium', \
               'Requirement too short (' + str(len(text.split())) + ' words) - likely missing constraints.'
    return 'No', 'None', 'N/A', 'No defect detected - requirement appears complete.'
